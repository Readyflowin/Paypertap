rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // =========================================================
    // CONSTANTS / HELPERS
    // =========================================================

    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn() && request.auth.token.admin == true;
    }

    function isOwner(sellerId) {
      return signedIn() && request.auth.uid == sellerId;
    }

    function phaseOneBookingAdvance() {
      return 20;
    }

    function isValidString(value) {
      return value is string && value.size() > 0 && value.size() <= 500;
    }

    function isValidOptionalString(value) {
      return value == null || (value is string && value.size() <= 1000);
    }

    function isValidOptionalShortString(value) {
      return value == null || (value is string && value.size() <= 200);
    }

    function isValidPhone(value) {
      return value is string && value.size() >= 8 && value.size() <= 15;
    }

    function isValidPincode(value) {
      return value is string && value.size() >= 4 && value.size() <= 10;
    }

    function isValidAmount(value) {
      return value is int && value >= 0 && value <= 1000000;
    }

    function isValidPositiveAmount(value) {
      return value is int && value > 0 && value <= 1000000;
    }

    function isPhaseOneBookingAmount(value) {
      return value is int && value == phaseOneBookingAdvance();
    }

    function isValidSellerStatus(value) {
      return value in ["active", "paused", "suspended"];
    }

    function isValidOnboardingStatus(value) {
      return value in [
        "auth_completed",
        "store_completed",
        "product_completed",
        "completed"
      ];
    }

    function isValidOnboardingStep(value) {
      return value in ["store", "product", "dashboard"];
    }

    function isValidProductStatus(value) {
      return value in ["draft", "open", "hold", "sold", "unpublished"];
    }

    function isPublicProductStatus(value) {
      return value in ["open", "hold", "sold"];
    }

    function isValidCheckoutStatus(value) {
      return value in [
        "started",
        "details_submitted",
        "payment_pending",
        "payment_created",
        "booking_paid",
        "payment_failed",
        "whatsapp_opened",
        "contacted",
        "remaining_paid",
        "confirmed",
        "sold",
        "abandoned",
        "expired",
        "cancelled",
        "released"
      ];
    }

    function isValidOrderStatus(value) {
      return value in [
        "new_booking",
        "contacted",
        "remaining_paid",
        "confirmed",
        "sold",
        "cancelled",
        "released"
      ];
    }

    function isValidPaymentStatus(value) {
      return value in ["pending", "booking_paid", "failed", "refunded"];
    }

    function isValidWhatsappStatus(value) {
      return value in ["not_opened", "opened", "message_sent"];
    }

    function isValidSlug(value) {
      return value is string && value.size() > 0 && value.size() <= 120;
    }

    function isValidCollectionName(value) {
      return value is string && value.size() > 0 && value.size() <= 80;
    }

    function isValidCollectionDescription(value) {
      return value is string && value.size() <= 500;
    }

    function isDurableImageUrl(value) {
      return value is string
        && value.size() > 0
        && value.size() <= 3000
        && value.matches('^https://.+');
    }

    function isValidOptionalUrl(value) {
      return value == null
        || (
          value is string
          && (
            value.size() == 0
            || (
              value.size() <= 3000
              && value.matches('^https://.+')
            )
          )
        );
    }

    function isValidImageListLoose(value) {
      return value is list && value.size() <= 3;
    }

    function hasValidProductVariants(data) {
      return (
          !("hasVariants" in data)
          || data.hasVariants is bool
        )
        && (
          !("variantOptions" in data)
          || (
            data.variantOptions is list
            && data.variantOptions.size() <= 2
          )
        )
        && (
          !("variants" in data)
          || (
            data.variants is list
            && data.variants.size() <= 100
          )
        )
        && (
          !("defaultVariantId" in data)
          || data.defaultVariantId == null
          || data.defaultVariantId is string
        );
    }

    function hasValidCollectionFields(data, storeId, collectionId) {
      return data.keys().hasOnly([
          "collectionId",
          "storeId",
          "name",
          "slug",
          "description",
          "createdAt",
          "updatedAt",
          "productCount"
        ])
        && data.collectionId == collectionId
        && data.storeId == storeId
        && isValidCollectionName(data.name)
        && isValidSlug(data.slug)
        && (
          !("description" in data)
          || isValidCollectionDescription(data.description)
        )
        && (
          !("createdAt" in data)
          || data.createdAt is timestamp
        )
        && (
          !("updatedAt" in data)
          || data.updatedAt is timestamp
        )
        && (
          !("productCount" in data)
          || (data.productCount is int && data.productCount >= 0)
        );
    }

    function storeExists(storeId) {
      return exists(/databases/$(database)/documents/stores/$(storeId));
    }

    function storeData(storeId) {
      return get(/databases/$(database)/documents/stores/$(storeId)).data;
    }

    function storeOwner(storeId) {
      return storeData(storeId).sellerId;
    }

    function isStoreOwner(storeId) {
      return signedIn()
        && storeExists(storeId)
        && storeOwner(storeId) == request.auth.uid;
    }

    function isPublishedStore(storeId) {
      return storeExists(storeId)
        && storeData(storeId).isPublished == true;
    }

    function productExists(productId) {
      return exists(/databases/$(database)/documents/products/$(productId));
    }

    function productData(productId) {
      return get(/databases/$(database)/documents/products/$(productId)).data;
    }

    function hasValidInventoryFields(data) {
      return data.inventoryQuantity is int
        && data.reservedQuantity is int
        && data.soldQuantity is int
        && data.inventoryQuantity >= 0
        && data.reservedQuantity >= 0
        && data.soldQuantity >= 0
        && data.reservedQuantity + data.soldQuantity <= data.inventoryQuantity;
    }

    function isOpenProductInPublishedStore(productId, storeId) {
      return productExists(productId)
        && productData(productId).storeId == storeId
        && productData(productId).status == "open"
        && productData(productId).inventoryQuantity is int
        && productData(productId).reservedQuantity is int
        && productData(productId).soldQuantity is int
        && productData(productId).inventoryQuantity
          - productData(productId).reservedQuantity
          - productData(productId).soldQuantity > 0
        && isPublishedStore(storeId);
    }

    function slugReservationExists(slug) {
      return exists(/databases/$(database)/documents/storeSlugs/$(slug));
    }

    function slugReservationData(slug) {
      return get(/databases/$(database)/documents/storeSlugs/$(slug)).data;
    }

    function isSlugReservedByOwner(slug) {
      return slugReservationExists(slug)
        && slugReservationData(slug).sellerId == request.auth.uid;
    }

    function hasOnlySellerMutableFields() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly([
        "name",
        "phone",
        "storeId",
        "status",
        "razorpayLinked",
        "profileImageUrl",
        "onboardingStatus",
        "onboardingStep",
        "updatedAt"
      ]);
    }

    function hasOnlyStoreMutableFields() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly([
        "storeName",
        "storeDescription",
        "description",
        "bio",
        "phone",
        "whatsappPhone",
        "logoUrl",
        "storeLogoUrl",
        "heroTitle",
        "heroSubtitle",
        "heroImageUrl",
        "instagramProfile",
        "instagramUrl",
        "themeId",
        "primaryColor",
        "secondaryColor",
        "accentColor",
        "isPublished",
        "bookingAdvanceAmount",
        "paymentMethods",
        "upiId",
        "googlePayNumber",
        "phonePeNumber",
        "paymentNote",
        "updatedAt"
      ]);
    }

    function hasOnlyOrderSellerMutableFields() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly([
        "orderStatus",
        "whatsappStatus",
        "sellerNotes",
        "updatedAt"
      ]);
    }

    function hasOnlyCustomerSellerMutableFields() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly([
        "tags",
        "notes",
        "updatedAt"
      ]);
    }

    function hasOnlyCheckoutSellerMutableFields() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly([
        "status",
        "whatsappOpened",
        "whatsappStatus",
        "sellerNotes",
        "updatedAt"
      ]);
    }

    function hasValidStoreCreateKeys(data) {
      return data.keys().hasOnly([
        "storeId",
        "sellerId",
        "storeSlug",
        "storeName",
        "storeDescription",
        "description",
        "bio",
        "phone",
        "whatsappPhone",
        "logoUrl",
        "storeLogoUrl",
        "heroTitle",
        "heroSubtitle",
        "heroImageUrl",
        "instagramProfile",
        "instagramUrl",
        "themeId",
        "primaryColor",
        "secondaryColor",
        "accentColor",
        "isPublished",
        "bookingAdvanceAmount",
        "paymentMethods",
        "upiId",
        "googlePayNumber",
        "phonePeNumber",
        "paymentNote",
        "createdAt",
        "updatedAt"
      ]);
    }

    function hasValidStoreOptionalFields(data) {
      return (
          !("storeDescription" in data)
          || isValidOptionalString(data.storeDescription)
        )
        && (
          !("description" in data)
          || isValidOptionalString(data.description)
        )
        && (
          !("bio" in data)
          || isValidOptionalString(data.bio)
        )
        && (
          !("phone" in data)
          || isValidOptionalString(data.phone)
        )
        && (
          !("whatsappPhone" in data)
          || isValidOptionalString(data.whatsappPhone)
        )
        && (
          !("logoUrl" in data)
          || isValidOptionalUrl(data.logoUrl)
        )
        && (
          !("storeLogoUrl" in data)
          || isValidOptionalUrl(data.storeLogoUrl)
        )
        && (
          !("heroTitle" in data)
          || isValidOptionalString(data.heroTitle)
        )
        && (
          !("heroSubtitle" in data)
          || isValidOptionalString(data.heroSubtitle)
        )
        && (
          !("heroImageUrl" in data)
          || isValidOptionalUrl(data.heroImageUrl)
        )
        && (
          !("instagramProfile" in data)
          || isValidOptionalString(data.instagramProfile)
        )
        && (
          !("instagramUrl" in data)
          || isValidOptionalString(data.instagramUrl)
        )
        && (
          !("themeId" in data)
          || data.themeId is string
        )
        && (
          !("primaryColor" in data)
          || data.primaryColor is string
        )
        && (
          !("secondaryColor" in data)
          || data.secondaryColor is string
        )
        && (
          !("accentColor" in data)
          || data.accentColor is string
        )
        && (
          !("bookingAdvanceAmount" in data)
          || isPhaseOneBookingAmount(data.bookingAdvanceAmount)
        )
        && (
          !("paymentMethods" in data)
          || data.paymentMethods is list
          || data.paymentMethods is map
        )
        && (
          !("upiId" in data)
          || isValidOptionalString(data.upiId)
        )
        && (
          !("googlePayNumber" in data)
          || isValidOptionalString(data.googlePayNumber)
        )
        && (
          !("phonePeNumber" in data)
          || isValidOptionalString(data.phonePeNumber)
        )
        && (
          !("paymentNote" in data)
          || isValidOptionalString(data.paymentNote)
        )
        && (
          !("createdAt" in data)
          || data.createdAt is timestamp
        )
        && (
          !("updatedAt" in data)
          || data.updatedAt is timestamp
        );
    }

    // =========================================================
    // SELLERS
    // =========================================================

    match /sellers/{sellerId} {
      allow create: if signedIn()
        && sellerId == request.auth.uid
        && request.resource.data.sellerId == request.auth.uid
        && request.resource.data.authUid == request.auth.uid
        && isValidSellerStatus(request.resource.data.status)
        && isValidOnboardingStatus(request.resource.data.onboardingStatus)
        && isValidOnboardingStep(request.resource.data.onboardingStep);

      allow read: if isOwner(sellerId) || isAdmin();

      allow update: if isOwner(sellerId)
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.authUid == resource.data.authUid
        && isValidSellerStatus(request.resource.data.status)
        && isValidOnboardingStatus(request.resource.data.onboardingStatus)
        && isValidOnboardingStep(request.resource.data.onboardingStep)
        && (
          !("profileImageUrl" in request.resource.data)
          || isValidOptionalUrl(request.resource.data.profileImageUrl)
        )
        && hasOnlySellerMutableFields();

      allow delete: if isAdmin();
    }

    // =========================================================
    // STORE SLUG RESERVATIONS
    // =========================================================

    match /storeSlugs/{slug} {
      allow get: if true;
      allow list: if false;

      allow create: if signedIn()
        && request.resource.data.slug == slug
        && request.resource.data.storeId == slug
        && request.resource.data.sellerId == request.auth.uid
        && !slugReservationExists(slug);

      allow update: if false;

      allow delete: if isAdmin();
    }

    // =========================================================
    // STORES
    // =========================================================

    match /stores/{storeId} {
      allow read: if resource.data.isPublished == true
        || isOwner(resource.data.sellerId)
        || isAdmin();

      allow create: if signedIn()
        && hasValidStoreCreateKeys(request.resource.data)
        && request.resource.data.sellerId == request.auth.uid
        && request.resource.data.storeId == storeId
        && request.resource.data.storeSlug == storeId
        && isSlugReservedByOwner(storeId)
        && isValidString(request.resource.data.storeName)
        && request.resource.data.isPublished is bool
        && request.resource.data.themeId is string
        && request.resource.data.primaryColor is string
        && request.resource.data.secondaryColor is string
        && request.resource.data.accentColor is string
        && hasValidStoreOptionalFields(request.resource.data);

      allow update: if isOwner(resource.data.sellerId)
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.storeId == resource.data.storeId
        && request.resource.data.storeSlug == resource.data.storeSlug
        && isValidString(request.resource.data.storeName)
        && request.resource.data.isPublished is bool
        && hasOnlyStoreMutableFields()
        && hasValidStoreOptionalFields(request.resource.data);

      allow delete: if isAdmin();
    }

    // =========================================================
    // STORE COLLECTIONS
    // =========================================================

    match /stores/{storeId}/collections/{collectionId} {
      allow read: if isPublishedStore(storeId)
        || isStoreOwner(storeId)
        || isAdmin();

      allow create: if isStoreOwner(storeId)
        && hasValidCollectionFields(request.resource.data, storeId, collectionId);

      allow update: if isStoreOwner(storeId)
        && resource.data.storeId == storeId
        && resource.data.collectionId == collectionId
        && request.resource.data.storeId == resource.data.storeId
        && request.resource.data.collectionId == resource.data.collectionId
        && hasValidCollectionFields(request.resource.data, storeId, collectionId);

      allow delete: if isStoreOwner(storeId) || isAdmin();
    }

    // =========================================================
    // THEMES
    // =========================================================

    match /themes/{themeId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    // =========================================================
    // PRODUCTS
    // Seller manages own products.
    // Images stay loose/max 3 as in old working rules.
    // Variants are only lightly capped.
    // =========================================================

    match /products/{productId} {
      allow read: if (
          isPublicProductStatus(resource.data.status)
          && isPublishedStore(resource.data.storeId)
        )
        || isOwner(resource.data.sellerId)
        || isAdmin();

      allow create: if signedIn()
        && request.resource.data.sellerId == request.auth.uid
        && request.resource.data.storeId is string
        && isStoreOwner(request.resource.data.storeId)
        && (
          !("productId" in request.resource.data)
          || request.resource.data.productId == productId
        )
        && isValidString(request.resource.data.title)
        && request.resource.data.description is string
        && request.resource.data.description.size() <= 5000
        && request.resource.data.category is string
        && request.resource.data.category.size() <= 120
        && isValidPositiveAmount(request.resource.data.price)
        && isPhaseOneBookingAmount(request.resource.data.bookingAdvanceAmount)
        && isValidAmount(request.resource.data.sellerCollectAmount)
        && request.resource.data.sellerCollectAmount ==
          request.resource.data.price - phaseOneBookingAdvance()
        && isValidProductStatus(request.resource.data.status)
        && request.resource.data.isFeatured is bool
        && request.resource.data.sortOrder is int
        && hasValidInventoryFields(request.resource.data)
        && (
          !("images" in request.resource.data)
          || isValidImageListLoose(request.resource.data.images)
        )
        && (
          !("imageUrl" in request.resource.data)
          || isValidOptionalUrl(request.resource.data.imageUrl)
        )
        && (
          !("thumbnailUrl" in request.resource.data)
          || isValidOptionalUrl(request.resource.data.thumbnailUrl)
        )
        && hasValidProductVariants(request.resource.data);

      allow update: if isOwner(resource.data.sellerId)
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.storeId == resource.data.storeId
        && (
          !("productId" in resource.data)
          || request.resource.data.productId == resource.data.productId
        )
        && request.resource.data.reservedQuantity == resource.data.reservedQuantity
        && request.resource.data.soldQuantity == resource.data.soldQuantity
        && isValidString(request.resource.data.title)
        && request.resource.data.description is string
        && request.resource.data.description.size() <= 5000
        && request.resource.data.category is string
        && request.resource.data.category.size() <= 120
        && isValidPositiveAmount(request.resource.data.price)
        && isPhaseOneBookingAmount(request.resource.data.bookingAdvanceAmount)
        && isValidAmount(request.resource.data.sellerCollectAmount)
        && request.resource.data.sellerCollectAmount ==
          request.resource.data.price - phaseOneBookingAdvance()
        && isValidProductStatus(request.resource.data.status)
        && request.resource.data.isFeatured is bool
        && request.resource.data.sortOrder is int
        && hasValidInventoryFields(request.resource.data)
        && (
          !("images" in request.resource.data)
          || isValidImageListLoose(request.resource.data.images)
        )
        && (
          !("imageUrl" in request.resource.data)
          || isValidOptionalUrl(request.resource.data.imageUrl)
        )
        && (
          !("thumbnailUrl" in request.resource.data)
          || isValidOptionalUrl(request.resource.data.thumbnailUrl)
        )
        && hasValidProductVariants(request.resource.data);

      allow delete: if isOwner(resource.data.sellerId) || isAdmin();
    }

    // =========================================================
    // CHECKOUT SESSIONS
    // Backend/Admin creates paid sessions only after verified Razorpay payment.
    // Browser cannot create booking_paid checkout sessions.
    // Seller can only update operational follow-up fields.
    // =========================================================

    match /checkoutSessions/{checkoutId} {
      allow create: if false;

      allow read: if isOwner(resource.data.sellerId) || isAdmin();

      allow update: if isOwner(resource.data.sellerId)
        && hasOnlyCheckoutSellerMutableFields()
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.storeId == resource.data.storeId
        && request.resource.data.productId == resource.data.productId
        && request.resource.data.productTitle == resource.data.productTitle
        && request.resource.data.productPrice == resource.data.productPrice
        && request.resource.data.bookingAdvanceAmount == resource.data.bookingAdvanceAmount
        && request.resource.data.sellerCollectAmount == resource.data.sellerCollectAmount
        && request.resource.data.buyerName == resource.data.buyerName
        && request.resource.data.buyerPhone == resource.data.buyerPhone
        && request.resource.data.buyerAddress == resource.data.buyerAddress
        && request.resource.data.buyerCity == resource.data.buyerCity
        && request.resource.data.buyerPincode == resource.data.buyerPincode
        && (
          !("buyerEmail" in resource.data)
          || request.resource.data.buyerEmail == resource.data.buyerEmail
        )
        && isValidCheckoutStatus(request.resource.data.status)
        && (
          !("whatsappOpened" in request.resource.data)
          || request.resource.data.whatsappOpened is bool
        )
        && (
          !("whatsappStatus" in request.resource.data)
          || isValidWhatsappStatus(request.resource.data.whatsappStatus)
        )
        && (
          !("sellerNotes" in request.resource.data)
          || request.resource.data.sellerNotes is string
        )
        && (
          !("updatedAt" in request.resource.data)
          || request.resource.data.updatedAt is timestamp
        );

      allow delete: if isAdmin();
    }

    // =========================================================
    // ORDERS / BOOKINGS
    // Backend creates orders after verified ₹20 booking payment.
    // Seller can read and update only operational status fields.
    // =========================================================

    match /orders/{orderId} {
      allow create: if false;

      allow read: if isOwner(resource.data.sellerId) || isAdmin();

      allow update: if isOwner(resource.data.sellerId)
        && request.resource.data.sellerId == resource.data.sellerId
        && request.resource.data.storeId == resource.data.storeId
        && request.resource.data.productId == resource.data.productId
        && request.resource.data.checkoutId == resource.data.checkoutId
        && request.resource.data.bookingAdvanceAmount == resource.data.bookingAdvanceAmount
        && request.resource.data.productPrice == resource.data.productPrice
        && request.resource.data.sellerCollectAmount == resource.data.sellerCollectAmount
        && isValidOrderStatus(request.resource.data.orderStatus)
        && isValidWhatsappStatus(request.resource.data.whatsappStatus)
        && hasOnlyOrderSellerMutableFields();

      allow delete: if isAdmin();
    }

    // =========================================================
    // CUSTOMERS / LEADS
    // Backend creates/updates customer records from paid bookings.
    // Seller can read and tag/note customers.
    // =========================================================

    match /customers/{customerId} {
      allow create: if false;

      allow read: if isOwner(resource.data.sellerId) || isAdmin();

      allow update: if isOwner(resource.data.sellerId)
        && request.resource.data.sellerId == resource.data.sellerId
        && hasOnlyCustomerSellerMutableFields();

      allow delete: if isAdmin();
    }

    // =========================================================
    // PAYMENTS
    // Backend/webhook only. Frontend never writes payment records.
    // =========================================================

    match /payments/{paymentId} {
      allow create, update, delete: if false;

      allow read: if isOwner(resource.data.sellerId) || isAdmin();
    }

    // =========================================================
    // EVENTS
    // Public can create limited analytics events only for published stores.
    // These events are not trusted payment/order records.
    // =========================================================

    match /events/{eventId} {
      allow create: if request.resource.data.keys().hasOnly([
          "sellerId",
          "storeId",
          "productId",
          "checkoutId",
          "eventType",
          "page",
          "source",
          "createdAt"
        ])
        && request.resource.data.sellerId is string
        && request.resource.data.storeId is string
        && isPublishedStore(request.resource.data.storeId)
        && request.resource.data.sellerId == storeOwner(request.resource.data.storeId)
        && request.resource.data.eventType in [
          "store_view",
          "product_click",
          "checkout_start",
          "details_submit",
          "payment_initiated",
          "booking_paid",
          "whatsapp_open"
        ]
        && (
          !("productId" in request.resource.data)
          || (
            request.resource.data.productId is string
            && productExists(request.resource.data.productId)
            && productData(request.resource.data.productId).storeId == request.resource.data.storeId
            && productData(request.resource.data.productId).sellerId == request.resource.data.sellerId
            && isPublicProductStatus(productData(request.resource.data.productId).status)
          )
        )
        && (
          !("checkoutId" in request.resource.data)
          || request.resource.data.checkoutId is string
        )
        && (
          !("page" in request.resource.data)
          || isValidOptionalString(request.resource.data.page)
        )
        && (
          !("source" in request.resource.data)
          || isValidOptionalString(request.resource.data.source)
        )
        && (
          !("createdAt" in request.resource.data)
          || request.resource.data.createdAt == request.time
        );

      allow read: if isOwner(resource.data.sellerId) || isAdmin();

      allow update, delete: if isAdmin();
    }
